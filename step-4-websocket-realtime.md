async sendMessage(text: string) {
  const message = createMessage(text);
  
  // Save locally first
  await storage.saveMessage(message);
  
  // Try WebSocket, queue if offline
  if (websocket.isConnected()) {
    websocket.send(message);
  } else {
    offlineQueue.enqueue(message);
  }
}
